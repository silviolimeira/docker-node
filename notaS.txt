# criando a aplicação node
$ npm init -y

# criar arquivos da aplicação
touch index.js
touch DockerFile

# instalar o express
$ npm install express

# montando a imagem
$ docker build -t sicemal/dockernode .

# rodando a aplicação
$ docker run -p 3000:3000 -d sicemal/dockernode
